## 04:03 UTC

**You:** Are aliens real

**Mentat:** The honest answer: we don't know.

We have no credible evidence of extraterrestrial life yet. The universe is vast — billions of galaxies, each with billions of stars — which makes the statistical argument for life elsewhere compelling. But statistics aren't data.

What we do know:
• Life emerged on Earth under specific conditions we're still mapping
• The building blocks (organic molecules, water) are common in space
• We've found potentially habitable exoplanets, but haven't detected biosignatures
• Our search methods (radio signals, spectroscopy) are still primitive relative to the scale we're scanning

The Fermi Paradox cuts the other way: if life is common, where is everyone? That silence is itself informative — either life is rarer than we think, or civilizations don't last long, or they're not broadcasting.

The recent UAP (Unidentified Aerial Phenomena) disclosures don't change this calculus. Unidentified ≠ extraterrestrial. Most resolve to mundane explanations once investigated properly.

The intellectually honest position: life beyond Earth probably exists somewhere in the cosmos. But "probably exists" and "we have evidence" are very different claims. We're still waiting for the evidence. 🔭

const categories = ["All", ...new Set(drugs.map(d => d.category))];

function shuffle(arr) {
  return [...arr].sort(() => Math.random() - 0.5);
}

function App() {
  const [mode, setMode] = React.useState("menu");
  const [category, setCategory] = React.useState("All");
  const [index, setIndex] = React.useState(0);
  const [flipped, setFlipped] = React.useState(false);
  const [quizDrugs, setQuizDrugs] = React.useState([]);
  const [quizIndex, setQuizIndex] = React.useState(0);
  const [quizMode, setQuizMode] = React.useState("use"); // use or dose
  const [options, setOptions] = React.useState([]);
  const [selected, setSelected] = React.useState(null);
  const [score, setScore] = React.useState(0);
  const [done, setDone] = React.useState(false);

  const filtered = category === "All" ? drugs : drugs.filter(d => d.category === category);

  function startFlashcards() {
    setIndex(0); setFlipped(false); setMode("flash");
  }

  function startQuiz(qm) {
    const deck = shuffle(filtered).slice(0, 15);
    setQuizDrugs(deck);
    setQuizIndex(0);
    setScore(0);
    setSelected(null);
    setDone(false);
    setQuizMode(qm);
    buildOptions(deck, 0, qm);
    setMode("quiz");
  }

  function buildOptions(deck, i, qm) {
    const correct = deck[i];
    const pool = drugs.filter(d => d.name !== correct.name);
    const wrong = shuffle(pool).slice(0, 3).map(d => qm === "use" ? d.use : d.dose);
    const answer = qm === "use" ? correct.use : correct.dose;
    setOptions(shuffle([answer, ...wrong]));
    setSelected(null);
  }

  function handleAnswer(opt) {
    if (selected !== null) return;
    setSelected(opt);
    const correct = quizMode === "use" ? quizDrugs[quizIndex].use : quizDrugs[quizIndex].dose;
    if (opt === correct) setScore(s => s + 1);
  }

  function nextQuestion() {
    const next = quizIndex + 1;
    if (next >= quizDrugs.length) { setDone(true); return; }
    setQuizIndex(next);
    buildOptions(quizDrugs, next, quizMode);
  }

  const card = filtered[index] || filtered[0];

  const styles = {
    app: { background: "#0a0a0a", minHeight: "100vh", color: "#ccc", fontFamily: "system-ui, sans-serif", padding: "20px", display: "flex", flexDirection: "column", alignItems: "center" },
    title: { color: "#c8860a", fontSize: "22px", fontWeight: "bold", marginBottom: "4px" },
    sub: { color: "#666", fontSize: "13px", marginBottom: "20px" },
    btn: { background: "#c8860a", color: "#000", border: "none", borderRadius: "8px", padding: "10px 22px", cursor: "pointer", fontWeight: "bold", fontSize: "14px", margin: "6px" },
    btnGhost: { background: "transparent", color: "#c8860a", border: "1px solid #c8860a", borderRadius: "8px", padding: "8px 18px", cursor: "pointer", fontSize: "13px", margin: "4px" },
    card: { background: flipped ? "#1a1200" : "#111", border: "1px solid #333", borderRadius: "14px", padding: "32px", width: "340px", minHeight: "180px", display: "flex", flexDirection: "column", alignItems: "center", justifyContent: "center", cursor: "pointer", transition: "background 0.2s", textAlign: "center" },
    tag: { background: "#c8860a22", color: "#c8860a", borderRadius: "20px", padding: "3px 12px", fontSize: "11px", marginBottom: "12px" },
    select: { background: "#111", color: "#ccc", border: "1px solid #333", borderRadius: "8px", padding: "6px 12px", marginBottom: "16px", fontSize: "13px" },
    optBtn: (opt) => ({
      display: "block", width: "100%", padding: "12px 16px", margin: "6px 0", borderRadius: "8px", border: "1px solid #333",
      background: selected === null ? "#111" : opt === (quizMode === "use" ? quizDrugs[quizIndex].use : quizDrugs[quizIndex].dose) ? "#1a3a1a" : selected === opt ? "#3a1a1a" : "#111",
      color: selected === null ? "#ccc" : opt === (quizMode === "use" ? quizDrugs[quizIndex].use : quizDrugs[quizIndex].dose) ? "#4caf50" : selected === opt ? "#f44336" : "#555",
      cursor: selected === null ? "pointer" : "default", textAlign: "left", fontSize: "13px"
    }),
  };

  if (mode === "menu") return (
    <div style={styles.app}>
      <div style={styles.title}>💊 Drug Memory</div>
      <div style={styles.sub}>Prescribing exam prep — 50 essential drugs</div>
      <select style={styles.select} value={category} onChange={e => setCategory(e.target.value)}>
        {categories.map(c => <option key={c}>{c}</option>)}
      </select>
      <div style={{ color: "#666", fontSize: "12px", marginBottom: "16px" }}>{filtered.length} drugs selected</div>
      <button style={styles.btn} onClick={startFlashcards}>🃏 Flashcards</button>
      <button style={styles.btn} onClick={() => startQuiz("use")}>🧠 Quiz — Indications</button>
      <button style={styles.btn} onClick={() => startQuiz("dose")}>💉 Quiz — Doses</button>
    </div>
  );

  if (mode === "flash") return (
    <div style={styles.app}>
      <div style={styles.title}>🃏 Flashcards</div>
      <div style={styles.sub}>{index + 1} / {filtered.length} — {category}</div>
      <select style={styles.select} value={category} onChange={e => { setCategory(e.target.value); setIndex(0); setFlipped(false); }}>
        {categories.map(c => <option key={c}>{c}</option>)}
      </select>
      <div style={styles.card} onClick={() => setFlipped(f => !f)}>
        <div style={styles.tag}>{card.category}</div>
        {!flipped ? (
          <div style={{ fontSize: "22px", fontWeight: "bold", color: "#fff" }}>{card.name}</div>
        ) : (
          <>
            <div style={{ fontSize: "13px", color: "#c8860a", marginBottom: "10px", fontWeight: "bold" }}>📋 Indication</div>
            <div style={{ fontSize: "14px", marginBottom: "14px" }}>{card.use}</div>
            <div style={{ fontSize: "13px", color: "#c8860a", marginBottom: "6px", fontWeight: "bold" }}>💉 Dose</div>
            <div style={{ fontSize: "13px", color: "#aaa" }}>{card.dose}</div>
          </>
        )}
        <div style={{ color: "#444", fontSize: "11px", marginTop: "16px" }}>tap to {flipped ? "hide" : "reveal"}</div>
      </div>
      <div style={{ marginTop: "16px" }}>
        <button style={styles.btnGhost} onClick={() => { setIndex(i => Math.max(0, i - 1)); setFlipped(false); }}>← Prev</button>
        <button style={styles.btnGhost} onClick={() => { setFlipped(false); setIndex(i => (i + 1) % filtered.length); }}>Next →</button>
      </div>
      <button style={{ ...styles.btnGhost, marginTop: "12px" }} onClick={() => setMode("menu")}>↩ Menu</button>
    </div>
  );

  if (mode === "quiz") return (
    <div style={styles.app}>
      {done ? (
        <>
          <div style={styles.title}>✅ Quiz Complete</div>
          <div style={{ fontSize: "48px", margin: "20px 0" }}>{score >= 12 ? "🏆" : score >= 8 ? "👍" : "📚"}</div>
          <div style={{ fontSize: "28px", color: "#c8860a", fontWeight: "bold" }}>{score} / {quizDrugs.length}</div>
          <div style={{ color: "#666", fontSize: "13px", margin: "8px 0 24px" }}>{score >= 12 ? "Excellent — you know your drugs." : score >= 8 ? "Solid — review the misses." : "More flashcard time needed."}</div>
          <button style={styles.btn} onClick={() => startQuiz(quizMode)}>Retry</button>
          <button style={styles.btnGhost} onClick={() => setMode("menu")}>↩ Menu</button>
        </>
      ) : (
        <>
          <div style={styles.title}>{quizMode === "use" ? "🧠 Indications Quiz" : "💉 Dose Quiz"}</div>
          <div style={styles.sub}>{quizIndex + 1} / {quizDrugs.length} · Score: {score}</div>
          <div style={{ background: "#111", border: "1px solid #333", borderRadius: "12px", padding: "24px", width: "340px", textAlign: "center", marginBottom: "16px" }}>
            <div style={styles.tag}>{quizDrugs[quizIndex].category}</div>
            <div style={{ fontSize: "20px", fontWeight: "bold", color: "#fff", marginBottom: "6px" }}>{quizDrugs[quizIndex].name}</div>
            <div style={{ color: "#666", fontSize: "12px" }}>{quizMode === "use" ? "What is the clinical indication?" : "What is the standard adult dose?"}</div>
          </div>
          <div style={{ width: "340px" }}>
            {options.map((opt, i) => (
              <button key={i} style={styles.optBtn(opt)} onClick={() => handleAnswer(opt)}>{opt}</button>
            ))}
          </div>
          {selected !== null && (
            <button style={{ ...styles.btn, marginTop: "12px" }} onClick={nextQuestion}>
              {quizIndex + 1 >= quizDrugs.length ? "See Results" : "Next →"}
            </button>
          )}
          <button style={{ ...styles.btnGhost, marginTop: "8px" }} onClick={() => setMode("menu")}>↩ Menu</button>
        </>
      )}
    </div>
  );
}

export default App;
