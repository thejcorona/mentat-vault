function getCategoryLabel(index) {
  for (const cat of categories) {
    if (index >= cat.range[0] && index <= cat.range[1]) return cat.label;
  }
  return "Other";
}

export default function App() {
  const [current, setCurrent] = React.useState(0);
  const [filter, setFilter] = React.useState("All");

  const allCats = ["All", ...categories.map(c => c.label)];
  const filtered = filter === "All" ? drugs.map((d, i) => ({ d, i }))
    : drugs.map((d, i) => ({ d, i })).filter(({ i }) => getCategoryLabel(i) === filter);

  const card = filtered[current];

  return (
    <div style={{
      minHeight: "100vh",
      background: "#0a0a0a",
      color: "#ccc",
      fontFamily: "system-ui, sans-serif",
      display: "flex",
      flexDirection: "column",
      alignItems: "center",
      padding: "32px 16px",
    }}>
      <div style={{ fontSize: 22, fontWeight: 700, color: "#c8860a", marginBottom: 4, letterSpacing: 1 }}>DRUG FLASHCARDS</div>
      <div style={{ fontSize: 13, color: "#555", marginBottom: 24 }}>Prescribing Exam Prep — Top 50</div>

      {/* Category Filter */}
      <div style={{ display: "flex", flexWrap: "wrap", gap: 8, justifyContent: "center", marginBottom: 28, maxWidth: 520 }}>
        {allCats.map(cat => (
          <button key={cat} onClick={() => { setFilter(cat); setCurrent(0); }} style={{
            padding: "5px 12px",
            borderRadius: 20,
            border: filter === cat ? "1.5px solid #c8860a" : "1.5px solid #333",
            background: filter === cat ? "#1f1600" : "#111",
            color: filter === cat ? "#c8860a" : "#666",
            cursor: "pointer",
            fontSize: 11,
            letterSpacing: 0.5,
          }}>{cat}</button>
        ))}
      </div>

      {card && <FlashCard drug={card.d} index={card.i} total={filtered.length} />}

      {/* Navigation */}
      <div style={{ display: "flex", gap: 16, marginTop: 28, alignItems: "center" }}>
        <button
          onClick={() => setCurrent(c => Math.max(0, c - 1))}
          disabled={current === 0}
          style={{
            padding: "10px 28px", borderRadius: 10, border: "1.5px solid #333",
            background: "#111", color: current === 0 ? "#333" : "#ccc",
            cursor: current === 0 ? "not-allowed" : "pointer", fontSize: 15,
          }}>← Prev</button>
        <span style={{ color: "#444", fontSize: 13 }}>{current + 1} / {filtered.length}</span>
        <button
          onClick={() => setCurrent(c => Math.min(filtered.length - 1, c + 1))}
          disabled={current === filtered.length - 1}
          style={{
            padding: "10px 28px", borderRadius: 10, border: "1.5px solid #333",
            background: "#111", color: current === filtered.length - 1 ? "#333" : "#ccc",
            cursor: current === filtered.length - 1 ? "not-allowed" : "pointer", fontSize: 15,
          }}>Next →</button>
      </div>

      <div style={{ marginTop: 20, fontSize: 12, color: "#444" }}>
        Shuffle tip: use category filters to drill by system
      </div>
    </div>
  );
}
